إعداد Supabase نفّذ هذه الأوامر أولاً

⚠️ قبل استخدام الصفحة، نفّذ الأوامر التالية في Supabase SQL Editor:

-- ════════════════════════════════════ -- 1. جدول contacts (زوار + مشتركون) -- ════════════════════════════════════ CREATE TABLE IF NOT EXISTS contacts ( id SERIAL PRIMARY KEY, "fullName" TEXT NOT NULL, phone TEXT NOT NULL, phone2 TEXT DEFAULT '', category TEXT DEFAULT '', source TEXT DEFAULT 'مباشر', notes TEXT DEFAULT '', stage TEXT DEFAULT 'زائر', "convertedAt" TIMESTAMP, "createdAt" TIMESTAMP DEFAULT NOW() ); -- ════════════════════════════════════ -- 2. جدول groups (مجموعات البرامج) -- ════════════════════════════════════ CREATE TABLE IF NOT EXISTS groups ( id SERIAL PRIMARY KEY, "programId" INT NOT NULL REFERENCES programs(id) ON DELETE CASCADE, name TEXT NOT NULL, capacity INT, instructor TEXT DEFAULT '', status TEXT DEFAULT 'active', "createdAt" TIMESTAMP DEFAULT NOW(), UNIQUE("programId", name) ); -- ════════════════════════════════════ -- 3. إضافة FKs لـ subscriptions (اختياري) -- ════════════════════════════════════ ALTER TABLE subscriptions ADD COLUMN IF NOT EXISTS "contactId" INT REFERENCES contacts(id), ADD COLUMN IF NOT EXISTS "groupId" INT REFERENCES groups(id); -- ════════════════════════════════════ -- 4. هجرة البيانات (اختياري) -- ════════════════════════════════════ -- نقل registrations → contacts (كـ زوار) INSERT INTO contacts ("fullName", phone, category, notes, stage, "createdAt") SELECT "fullName", phone, category, COALESCE(notes,''), 'زائر', "createdAt" FROM registrations ON CONFLICT DO NOTHING; -- نقل subscriptions → contacts (كـ مشتركون، distinct بالجوال) INSERT INTO contacts ("fullName", phone, category, stage, "convertedAt", "createdAt") SELECT DISTINCT ON (phone) "studentName", phone, category, 'مشترك', "startDate"::timestamp, "createdAt" FROM subscriptions ORDER BY phone, "createdAt" ASC ON CONFLICT DO NOTHING;
